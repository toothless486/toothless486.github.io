### 코드
```c++
#include <iostream>
#include <stack>
#include  <vector>
using namespace std;
int main()
{
    int n;
    cin >> n;

    vector<long long> arr(n+1,  0);
    for(int i=1;i <=n; i++)
    {
        cin >> arr[i];
    }

    stack<pair<long long, int>> s;
    long long result = 0;

    for(int i=1; i<=n; i++)
    {
        int a = arr[i];
        pair<long long, int> input = {a, 1};

        while(!s.empty())
        {
            if(s.top().first <= a)
            {
                result += s.top().second;
                if(s.top().first == a)
                {
                    input.second += s.top().second;
                }
                s.pop();
            }
            else
            {
                break;
            }
        }

        if(!s.empty())
        {
            result++;
        }
        s.push(input);
    }
    cout  << result;
    return 0;
}
```
